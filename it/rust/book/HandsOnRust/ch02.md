**将输入处理逻辑移入函数**

```rust
 use std::io::stdin;
 
 fn main() {
     println!("Hello, world!");
     let name = what_is_your_name();
     println!("Hello, {}", name);
 }
 
 fn what_is_your_name() -> String {
     let mut your_name = String::new();
 
     stdin()
         .read_line(&mut your_name)
         .expect("Faild to read line");
 
     your_name.trim().to_lowercase()
 }
```
